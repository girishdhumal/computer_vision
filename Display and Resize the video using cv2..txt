import cv2
vidcap = cv2.VideoCapture('1-second-video.mp4')
success,image = vidcap.read()
count = 0;
while success:
  success,image = vidcap.read()
  resized = cv2.resize(image, (400, 600), interpolation=cv2.INTER_LINEAR)
  cv2.imwrite("%03d.jpg" % count, resized)
  if cv2.waitKey(10) == 27:
      break
  count += 1
