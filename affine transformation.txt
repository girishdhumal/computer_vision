import cv2
import matplotlib.pyplot as plt
import numpy as np
def plot_cv_img(input_img):
  plt.imshow(cv2.cvtColor(input_img,cv2.COLOR_BGR2RGB))
  plt.show()
def main():
  img=cv2.imread("picture.jpg")
  pts1 = np.float32([[50,50],[200,50],[50,200]])
  pts2 = np.float32([[10,100],[200,50],[1000,250]])
  affine_tr = cv2.getAffineTransform(pts1,pts2)
  transform = cv2.warpAffine(img,affine_tr,(img.shape[1]*2,img.shape[0]*2))
  plot_cv_img(transform)
main()
