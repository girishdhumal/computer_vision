from PIL.ImageFilter import (BLUR, CONTOUR, DETAIL, EMBOSS, SMOOTH, SHARPEN)
from PIL import Image
img = Image.open('picture.jpg')
img1 = img.filter(BLUR)
img1.save('BLurred.jpg')
img1.show(img1)
img2 = img.filter(CONTOUR)
img2.save('contour.jpg')
img2.show()
img3= img.filter(DETAIL)
img3.save('detail.jpg')
img3.show()
img4= img.filter(EMBOSS)
img4.save('emboss.jpg')
img4.show()
img5= img.filter(SMOOTH)
img5.save('smooth.jpg')
img5.show()
img6= img.filter(SHARPEN)
img6.save('sharpen.jpg')
img6.show()
