#Aim :- Demonstrating some image filter that can be applied to computer vision
import cv2
#load and read the image from file.
print('Siddhi Atre, Roll no- 04')
img=cv2.imread("picture.jpg")
cv2.imshow('img',img)
HSV=cv2.cvtColor(img,cv2.COLOR_BGR2HSV)
cv2.imshow('hsvimg',HSV)
cv2.waitKey(0)
#Aim :- Demonstrating some image filter that can be applied to computer vision
cv2.destroyAllWindows()
gray=cv2.cvtColor(img,cv2.COLOR_BGR2GRAY)
cv2.imshow('grayimg',gray)
cv2.waitKey(0)
cv2.destroyAllWindows()
