import cv2
import numpy as np
square=np.zeros((800,800,3),dtype='uint8')
#circle
cv2.circle(square,(600,600),45,(255,127,80),thickness=3)
#rectangle
cv2.rectangle(square,(0,0),(300,300),(0, 0, 200),thickness=cv2.FILLED)
#line
cv2.line(square,(500,200),(200,500),(205,92,92),5)
cv2.imshow("square",square)
cv2.waitKey(0)
cv2.destroyAllwindows()
